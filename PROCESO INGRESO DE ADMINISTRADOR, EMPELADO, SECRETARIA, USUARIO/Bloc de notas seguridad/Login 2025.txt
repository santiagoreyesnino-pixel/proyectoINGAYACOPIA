  <script>
    function iniciarSesion() {
      const usuario = document.getElementById("usuario").value.trim().toLowerCase();
      const contraseña = document.getElementById("contraseña").value;

      // Accesos simulados
      const usuarios = {
        'admin': { pass: '12345', rol: 'panel administrador.html' },
        'empleado': { pass: '1234', rol: 'empleado.html' },
        'secretaria': { pass: '123', rol: 'secretaria.html' }
      };

      if (usuarios[usuario] && usuarios[usuario].pass === contraseña) {
        window.open(usuarios[usuario].rol, '_blank'); // Abre en nueva pestaña
      } else {
        alert("Credenciales incorrectas. Intenta nuevamente.");
      }
    }
  </script>
